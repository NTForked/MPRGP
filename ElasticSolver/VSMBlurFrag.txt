#define NR_BLUR_MAX 20

uniform sampler2D motions;
uniform vec2 scaleU;

uniform float offB[NR_BLUR_MAX];
uniform float coefB[NR_BLUR_MAX];
uniform int nrB;

void main()
{
	vec4 color=vec4(0.0);
	for(int i=0;i<nrB;i++)
		color+=texture2D(motions,gl_TexCoord[0].xy+scaleU*offB[i])*coefB[i];
	gl_FragColor=color;
}
